% ─────────────────────────────────────────────────────────────────────────────
\subsection{Motivation}

In a search engine, certain terms are searched much more often than others. If we
store keys in a BST, the search order depends on the tree's structure. Placing the
most frequent keys near the root minimizes the average cost of a search. The OBST
problem formalizes this idea.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Problem Definition}

\begin{definition}{Optimal Binary Search Tree}{obst}
\textbf{Input:} a sorted sequence of $n$ distinct keys
$K = \langle k_1 < k_2 < \cdots < k_n \rangle$, with a search probability 
$p_i > 0$ for each $k_i$ (where $\sum_{i=1}^n p_i = 1$).

\textbf{Objective:} construct a BST $T$ on these keys that minimizes the expected
search cost:
\[
  E[\text{search cost in } T]
  = \sum_{i=1}^{n} (\depth_T(k_i) + 1)\, p_i
  = 1 + \sum_{i=1}^{n} \depth_T(k_i)\cdot p_i.
\]
\end{definition}

The actual cost of searching for $k_i$ is $\depth_T(k_i) + 1$ (the number of nodes
examined, including the root).

\begin{remark}{Key Observations}{obsobs}
\begin{itemize}[noitemsep]
  \item The OBST is \emph{not} necessarily the tree with minimum height.
  \item The key with the highest probability is \emph{not} necessarily at the root.
  \item The number of BSTs with $n$ nodes is exponential (Catalan numbers $\sim 4^n/n^{3/2}$),
        so brute force is impractical.
\end{itemize}
\end{remark}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Optimal Substructure}

A BST is built by choosing a root $k_r$, then recursively building the left
subtree on $k_i, \ldots, k_{r-1}$ and the right subtree on $k_{r+1}, \ldots, k_j$.

\begin{theorem}{Optimal Substructure of OBST}{obstopt}
If we build an optimal BST on $k_i, \ldots, k_j$ with root $k_r$, then the
left and right subtrees must also be OBSTs (on $k_i, \ldots, k_{r-1}$ and
$k_{r+1}, \ldots, k_j$ respectively).
\end{theorem}

\textbf{Cost analysis.} Letting $w(i,j) = \sum_{\ell=i}^{j} p_\ell$ be the sum
of probabilities over $k_i, \ldots, k_j$, we observe that when a subtree becomes
the left or right child of the root, the depth of each of its nodes increases by 1.
Therefore:

\[
  E[\text{cost of subtree } k_i \ldots k_j]
  = p_r + \bigl(w(i,r-1) + E[\text{cost of left}]\bigr)
        + \bigl(w(r+1,j) + E[\text{cost of right}]\bigr).
\]

By regrouping, we get:
\[
  E[\text{cost}] = p_r + w(i,r-1) + E[\text{cost of left}] + w(r+1,j) + E[\text{cost of right}]
                 = w(i,j) + E[\text{cost of left}] + E[\text{cost of right}].
\]

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Recursive Formulation}

Let us define $e[i,j]$ as the expected search cost of the OBST built on
$k_i, \ldots, k_j$, and $w(i,j) = \sum_{\ell=i}^{j} p_\ell$.

\[
  e[i, j] =
  \begin{cases}
    0 & \text{if } i = j + 1 \quad (\text{empty subtree}), \\[6pt]
    \displaystyle\min_{i \le r \le j}
      \bigl\{ e[i, r-1] + e[r+1, j] + w(i,j) \bigr\}
    & \text{if } i \le j.
  \end{cases}
\]

We want to calculate $e[1, n]$.

\begin{remark}{Link with the Sum of Probabilities}{wlink}
The term $w(i,j) = \sum_{\ell=i}^{j} p_\ell$ can be precalculated 
incrementally: $w(i,j) = w(i,j-1) + p_j$, which avoids recalculating it at each
step.
\end{remark}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Bottom-up Algorithm}

We fill tables $e$ and $r$ (optimal roots) by increasing chain length 
($\ell = j - i + 1$ from 1 to $n$):

\begin{algorithm}[H]
\caption{Optimal-BST($p$, $n$)}
\begin{algorithmic}[1]
\State Create tables $e[1\ldots n+1,\; 0\ldots n]$,
       $w[1\ldots n+1,\; 0\ldots n]$ and $r[1\ldots n,\; 1\ldots n]$
\For{$i = 1$ \textbf{to} $n+1$}
    \State $e[i, i-1] \leftarrow 0$     \AlgComment{empty subtrees}
    \State $w[i, i-1] \leftarrow 0$
\EndFor
\For{$\ell = 1$ \textbf{to} $n$}        \AlgComment{chain length}
    \For{$i = 1$ \textbf{to} $n - \ell + 1$}
        \State $j \leftarrow i + \ell - 1$
        \State $e[i,j] \leftarrow \infty$
        \State $w[i,j] \leftarrow w[i, j-1] + p_j$
        \For{$r = i$ \textbf{to} $j$}   \AlgComment{try each root}
            \State $t \leftarrow e[i, r-1] + e[r+1, j] + w[i,j]$
            \If{$t < e[i,j]$}
                \State $e[i,j] \leftarrow t$
                \State $\mathit{root}[i,j] \leftarrow r$  \AlgComment{store optimal root}
            \EndIf
        \EndFor
    \EndFor
\EndFor
\State \Return $e$ and $\mathit{root}$
\end{algorithmic}
\end{algorithm}

\begin{complexity}{Optimal-BST}{obstcplx}
\begin{itemize}[noitemsep]
  \item Table $e$ contains $O(n^2)$ entries.
  \item Each entry $e[i,j]$ requires $O(n)$ to test all roots $r$.
  \item \textbf{Total complexity: $\Theta(n^3)$ in time, $\Theta(n^2)$ in space.}
\end{itemize}
\end{complexity}

\textbf{Calculation of $e[1,2]$ for illustration:}
$w(1,2) = p_1 + p_2 = 0.45$. We test:
\begin{itemize}[noitemsep]
  \item $r = 1$: $e[1,0] + e[2,2] + 0.45 = 0 + 0.20 + 0.45 = 0.65$

\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Reconstructing the Optimal Tree}

Table $\mathit{root}[i,j]$ stores the chosen root at each step. We
reconstruct the tree recursively:

\begin{algorithm}[H]
\caption{Build-Optimal-BST($\mathit{root}$, $i$, $j$, $\mathit{parent}$)}
\begin{algorithmic}[1]
\If{$i > j$}
    \State \Return \AlgComment{empty subtree}
\EndIf
\State $r \leftarrow \mathit{root}[i,j]$
\State \textbf{print} ``$k_r$ is \{root / left child / right child\} of $\mathit{parent}$''
\State \Call{Build-Optimal-BST}{$\mathit{root}$, $i$, $r-1$, $k_r$}   \AlgComment{left subtree}
\State \Call{Build-Optimal-BST}{$\mathit{root}$, $r+1$, $j$, $k_r$}   \AlgComment{right subtree}
\end{algorithmic}
\end{algorithm}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Summary: OBST}

\begin{tcolorbox}[colback=lightblue, colframe=keyblue, fonttitle=\bfseries,
                  title={Summary DP – Optimal Binary Search Tree}]
\begin{itemize}[noitemsep]
  \item \textbf{Choice}: which key $k_r$ to place at the root of the subtree
        $k_i, \ldots, k_j$.
  \item \textbf{Optimal substructure}: the left and right subtrees must
        themselves be OBSTs.
  \item \textbf{Recurrence}:
    \[
      e[i,j] =
      \begin{cases}
        0 & i = j+1,\\
        \displaystyle\min_{i \le r \le j}
          \bigl\{e[i,r-1] + e[r+1,j] + w(i,j)\bigr\} & i \le j.
      \end{cases}
    \]
  \item \textbf{Complexity}: $\Theta(n^3)$ in time, $\Theta(n^2)$ in space.
  \item \textbf{Reconstruction}: table $\mathit{root}[i,j]$ + recursive calls.
\end{itemize}
\end{tcolorbox}
