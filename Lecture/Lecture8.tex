\subsection{Definition and Fundamental Property}

\begin{definition}{Binary Search Tree (BST)}{bst}
A \textbf{BST} is a binary tree $T$ where each node $x$ satisfies:
\begin{itemize}[noitemsep]
  \item If $y$ is in the \emph{left subtree} of $x$:
        $y.\mathit{key} < x.\mathit{key}$.
  \item If $y$ is in the \emph{right subtree} of $x$:
        $y.\mathit{key} \ge x.\mathit{key}$.
\end{itemize}
The \textbf{height} $h$ is the number of edges on the longest root–leaf path.
\end{definition}

Search, insertion, deletion, min, max, successor, predecessor:
all in $O(h)$. For a balanced tree: $h = O(\log n)$.
\end{complexity}

\subsection{Queries on a BST}

\subsubsection{Search}

\begin{algorithm}[H]
\caption{BST-Search($T$, $k$)}
\begin{algorithmic}[1]
\State $x \leftarrow T.\mathit{root}$
\While{$x \ne \textsc{NIL}$ \textbf{and} $k \ne x.\mathit{key}$}
  \If{$k < x.\mathit{key}$}  \State $x \leftarrow x.\mathit{left}$
  \Else                       \State $x \leftarrow x.\mathit{right}$
  \EndIf
\EndWhile
\State \Return $x$
\end{algorithmic}
\end{algorithm}

\subsubsection{Minimum, Maximum, Successor}

\begin{theorem}{Locating min/max}{minmax}
\begin{itemize}[noitemsep]
  \item The \textbf{minimum} is the leftmost node.
  \item The \textbf{maximum} is the rightmost node.
\end{itemize}
\end{theorem}

The \textbf{successor} of $x$ (smallest key $> x.\mathit{key}$) is found:
\begin{enumerate}[noitemsep]
  \item If $x$ has a right subtree: it is the \emph{minimum} of the right subtree.
  \item Otherwise: go up to the first ancestor $y$ such that $x$ is in the \emph{left} 
        subtree of $y$.
\end{enumerate}

\subsection{BST Traversal}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{llll}
\toprule
\textbf{Traversal} & \textbf{Order} & \textbf{Result (ex.)} & \textbf{Utility} \\
\midrule
Inorder     & left, root, right & $1,2,\ldots,n$ (sorted!) & Sorting \\
Preorder    & root, left, right & root first & Copying \\
Postorder   & left, right, root & root last & Deletion \\
\bottomrule
\end{tabular}
\end{center}

\begin{remark}{Inorder Traversal and Sorting}{inorder}
The inorder traversal of a BST always produces keys in \textbf{increasing order}
(a direct property of the BST definition). Complexity: $\Theta(n)$.
\end{remark}

\end{remark}

\subsection{Paradigm and Central Idea}

\begin{definition}{Dynamic Programming (DP)}{dp}
\textbf{Dynamic programming} is an algorithmic paradigm (not a language feature). 
Its principle:
\begin{center}
  \textbf{Store already performed calculations to avoid repeating them.}
\end{center}
This allows transforming exponential algorithms into polynomial algorithms.
\end{definition}

A DP algorithm relies on two properties:
\begin{enumerate}
  \item \textbf{Optimal substructure}: the optimal solution to a problem contains the 
        optimal solutions of its subproblems.
  \item \textbf{Overlapping subproblems}: a naive recursive algorithm recalculates 
        the same subproblems an exponential number of times.
\end{enumerate}

\textbf{Two implementation strategies:}
\begin{itemize}
  \item \textbf{Top-down with memoization}: recursion + results table.
  \item \textbf{Bottom-up}: filling the table from small subproblems toward large ones.
\end{itemize}

\subsection{Rod Cutting}

\begin{definition}{Rod Cutting Problem}{rodcut}
\textbf{Input:} a rod of length $n$ and a table of prices $p_i$ for a rod of 
length $i$ ($1 \le i \le n$). \\
\textbf{Objective:} find the cutting that maximizes total revenue.
\end{definition}

There are $2^{n-1}$ possible cuttings: brute force is unfeasible.

\subsubsection{Optimal Substructure and Recurrence}

\begin{theorem}{Optimal Substructure}{rodthm}
If the first optimal cut is after $i$ units, and if
$s_1, \ldots, s_k$ is an optimal cutting of the remaining rod of length $n-i$,
then $i, s_1, \ldots, s_k$ is optimal for the rod of length $n$.
\end{theorem}

Let $r(n)$ be the optimal revenue for a rod of length $n$:
\[
  r(n) =
  \begin{cases}
    0 & \text{if } n = 0, \\
    \displaystyle\max_{1 \le i \le n} \bigl\{ p_i + r(n-i) \bigr\} & \text{if } n \ge 1.
  \end{cases}
\]

\begin{complexity}{DP for Rod Cutting}{rcplx}
\begin{itemize}[noitemsep]
  \item Top-down with memoization: $O(n^2)$ (subproblem graph).
  \item Bottom-up: $O(n^2)$.
\end{itemize}
\end{complexity}

\subsubsection{Reconstruction of the Solution}

We maintain an array $s[j]$ storing the first optimal cut for $r(j)$.
Then, we trace-back: cut $s[n]$, then solve for $n - s[n]$.

\[
\begin{array}{cccccccccc}
\toprule
i    & 0 & 1 & 2 & 3 & 4  & 5  & 6  & 7  & 8  \\
\midrule
r[i] & 0 & 1 & 5 & 8 & 10 & 13 & 17 & 18 & 22 \\
s[i] & 0 & 1 & 2 & 3 &  2 &  2 &  6 &  1 &  2 \\
\bottomrule
\end{array}
\]

\subsection{Coin Change Problem}

\begin{definition}{Coin Change}{change}
\textbf{Input:} $n$ denominations $0 < w_1 < \cdots < w_n$ and a target amount $W$. \\
\textbf{Output:} minimum number of coins to reach exactly $W$.
\end{definition}

DP Recurrence:
\[
  c(w) =
  \begin{cases}
    0 & \text{if } w = 0, \\
    \displaystyle\min_{\substack{1 \le j \le n \\ w_j \le w}} \bigl\{1 + c(w - w_j)\bigr\}
       & \text{if } w \ge 1.
  \end{cases}
\]

\textbf{Example:} $w_1=1, w_2=2, w_3=5$, $W=8$.
Optimal: 1 coin of each $\Rightarrow$ 3 coins in total.

Bottom-up filling from $w=0$ to $W$: complexity $O(nW)$.

\subsection{General Recipe for DP}

\begin{tcolorbox}[colback=lightblue, colframe=keyblue,
                  title={\textbf{Design Steps for a DP Algorithm}},
                  fonttitle=\bfseries]
\begin{enumerate}
  \item \textbf{Identify choices and optimal substructure}: a choice reduces
        the problem to smaller subproblems, whose optimal solutions
        compose the global solution.
  \item \textbf{Write a recurrence} for the optimal value based on the
        subproblems.
  \item \textbf{Avoid recalculations} via memoization (top-down) or table filling
        (bottom-up).
  \item \textbf{(Optional) Reconstruct the solution} by storing choices in an
        auxiliary table and tracing back.
\end{enumerate}
\end{tcolorbox}
