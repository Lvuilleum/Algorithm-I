\subsection{Définitions et représentations}

\begin{definition}{Graphe}{graph}
Un \textbf{graphe} $G = (V, E)$ est constitué d'un ensemble de \textbf{sommets}
$V$ et d'un ensemble d'\textbf{arêtes} $E$ contenant des paires (ordonnées ou non)
de sommets. Un graphe peut être \emph{non orienté}, \emph{orienté},
\emph{pondéré sur les sommets}, \emph{pondéré sur les arêtes}, etc.
\end{definition}

\subsubsection{Listes d'adjacence}

On maintient un tableau $\mathit{Adj}$ de $|V|$ listes : la liste de $u$ contient
tous les sommets $v$ tels que $(u, v) \in E$. On note $G.\mathit{Adj}[u]$ dans les
algorithmes.

\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{lll}
\toprule
\textbf{Opération} & \textbf{Coût} & \\
\midrule
Espace & $\Theta(V + E)$ & compact pour les graphes creux \\
Lister les voisins de $u$ & $\Theta(\deg(u))$ & \\
Tester $(u,v) \in E$ & $O(\deg(u))$ & \\
\bottomrule
\end{tabular}
\end{center}

\subsubsection{Matrice d'adjacence}

Une matrice $|V| \times |V|$ où $a_{ij} = 1$ si $(i,j) \in E$, 0 sinon.

\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{lll}
\toprule
\textbf{Opération} & \textbf{Coût} & \\
\midrule
Espace & $\Theta(V^2)$ & dense, simple à implémenter \\
Lister les voisins de $u$ & $\Theta(V)$ & \\
Tester $(u,v) \in E$ & $\Theta(1)$ & \\
\bottomrule
\end{tabular}
\end{center}

\begin{remark}{Choix de la représentation}{graphrep}
Les deux représentations peuvent être étendues aux graphes pondérés.
Pour les graphes \emph{creux} ($|E| \ll |V|^2$), les listes d'adjacence sont
préférables en espace. La matrice est avantageuse quand on teste souvent
l'appartenance d'une arête.
\end{remark}

\subsection{Parcours en largeur (BFS)}

\begin{definition}{Breadth-First Search (BFS)}{bfs}
\textbf{Entrée :} graphe $G = (V,E)$ (orienté ou non), sommet source $s \in V$.

\textbf{Sortie :} pour tout sommet $v$, $v.d$ = distance minimale (en nombre
d'arêtes) de $s$ à $v$.
\end{definition}

\textbf{Idée :} émettre une vague depuis $s$. Elle atteint d'abord tous les
sommets à distance~1, puis à distance~2, etc. On utilise une \emph{file} (FIFO).

\subsubsection{Pseudocode}

\begin{algorithm}[H]
\caption{BFS($G$, $s$)}
\begin{algorithmic}[1]
\For{chaque sommet $u \in G.V \setminus \{s\}$}
    \State $u.d \leftarrow \infty$
\EndFor
\State $s.d \leftarrow 0$
\State $Q \leftarrow \emptyset$;\; \textsc{Enqueue}$(Q, s)$
\While{$Q \ne \emptyset$}
    \State $u \leftarrow \textsc{Dequeue}(Q)$
    \For{chaque $v \in G.\mathit{Adj}[u]$}
        \If{$v.d = \infty$}           \AlgComment{$v$ non encore découvert}
            \State $v.d \leftarrow u.d + 1$
            \State \textsc{Enqueue}$(Q, v)$
        \EndIf
    \EndFor
\EndWhile
\end{algorithmic}
\end{algorithm}

Les distances finales sont bien les distances dans le graphe.

\subsubsection{Analyse}

\begin{complexity}{BFS}{bfscplx}
\begin{itemize}[noitemsep]
  \item Chaque sommet est mis en file au plus une fois : $O(V)$.
  \item Chaque arête est examinée au plus une fois (orienté) ou deux fois
        (non orienté) : $O(E)$.
  \item \textbf{Complexité totale : $O(V + E)$.}
\end{itemize}
\end{complexity}

\textbf{Note :} BFS ne visite pas forcément tous les sommets (seulement ceux
atteignables depuis $s$). On peut enregistrer l'\emph{arbre BFS} en mémorisant
l'arête qui a découvert chaque sommet ($v.\pi \leftarrow u$).

\subsection{Parcours en profondeur (DFS)}

\begin{definition}{Depth-First Search (DFS)}{dfs}
\textbf{Entrée :} graphe $G = (V,E)$, orienté ou non.

\textbf{Sortie :} pour tout sommet $v$, deux horodatages :
$v.d$ (temps de \emph{découverte}) et $v.f$ (temps de \emph{fin}).
\end{definition}

\textbf{Idée :} explorer chaque arête à fond avant de revenir en arrière.
Dès qu'un sommet est découvert, on explore depuis lui (contrairement à BFS
qui explore d'abord les proches). On repart d'un nouveau sommet non découvert
si nécessaire.

\subsubsection{Couleurs des sommets}

Durant l'exécution, chaque sommet a une couleur :
\begin{itemize}[noitemsep]
  \item \textbf{BLANC} : non découvert.
  \item \textbf{GRIS} : découvert mais non terminé (exploration en cours depuis lui).
  \item \textbf{NOIR} : terminé (tout ce qui est atteignable depuis lui a été exploré).
\end{itemize}

\subsubsection{Pseudocode}

\begin{algorithm}[H]
\caption{DFS($G$)}
\begin{algorithmic}[1]
\For{chaque sommet $u \in G.V$}
    \State $u.\mathit{color} \leftarrow \text{BLANC}$
\EndFor
\State $\mathit{time} \leftarrow 0$
\For{chaque sommet $u \in G.V$}
    \If{$u.\mathit{color} = \text{BLANC}$}
        \State \Call{DFS-Visit}{$G$, $u$}
    \EndIf
\EndFor
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{DFS-Visit($G$, $u$)}
\begin{algorithmic}[1]
\State $\mathit{time} \leftarrow \mathit{time} + 1$
\State $u.d \leftarrow \mathit{time}$            \AlgComment{horodatage découverte}
\State $u.\mathit{color} \leftarrow \text{GRIS}$
\For{chaque $v \in G.\mathit{Adj}[u]$}
    \If{$v.\mathit{color} = \text{BLANC}$}
        \State \Call{DFS-Visit}{$G$, $v$}
    \EndIf
\EndFor
\State $u.\mathit{color} \leftarrow \text{NOIR}$
\State $\mathit{time} \leftarrow \mathit{time} + 1$
\State $u.f \leftarrow \mathit{time}$            \AlgComment{horodatage fin}
\end{algorithmic}
\end{algorithm}

\subsubsection{Analyse}

\begin{complexity}{DFS}{dfscplx}
\begin{itemize}[noitemsep]
  \item Chaque sommet est découvert exactement une fois : $\Theta(V)$.
  \item Chaque arête est examinée exactement une fois (orienté) ou deux fois
        (non orienté) : $\Theta(E)$.
  \item \textbf{Complexité totale : $\Theta(V + E)$.}
\end{itemize}
\end{complexity}

DFS produit une \textbf{forêt DFS} : une ou plusieurs arborescences, chacune
issue d'un appel de \textsc{DFS-Visit} dans la boucle principale.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Classification des arêtes (DFS)}

Dans un DFS, chaque arête $(u,v)$ peut être classifiée selon les horodatages :

\begin{center}
\renewcommand{\arraystretch}{1.35}
\begin{tabular}{lll}
\toprule
\textbf{Type} & \textbf{Définition} & \textbf{Condition} \\
\midrule
\textbf{Arête d'arbre} & dans la forêt DFS & $v$ était BLANC quand $(u,v)$ exploré \\
\textbf{Arête arrière} & $(u,v)$ : $u$ est descendant de $v$ & $v.d < u.d < u.f < v.f$ \\
\textbf{Arête avant}   & $(u,v)$ : $v$ est descendant de $u$, \newline mais pas via arbre & $u.d < v.d < v.f < u.f$ \\
\textbf{Arête croisée} & toute autre arête & $v.f < u.d$ \\
\bottomrule
\end{tabular}
\end{center}

\begin{remark}{Graphes non orientés}{undirected}
Dans le DFS d'un graphe \emph{non orienté}, on n'obtient que des arêtes d'arbre
et des arêtes arrière — jamais d'arêtes avant ni d'arêtes croisées. En effet, si
$(u,v)$ est examiné depuis $u$ et que $v$ est déjà GRIS ou NOIR, alors $v$ est
nécessairement un ancêtre de $u$ dans l'arbre courant, ce qui donne une arête
arrière.
\end{remark}

\subsection{Résumé : BFS et DFS}

\begin{center}
\renewcommand{\arraystretch}{1.45}
\begin{tabular}{lll}
\toprule
& \textbf{BFS} & \textbf{DFS} \\
\midrule
Structure utilisée & File (FIFO) & Pile (récursion) \\
Ordre d'exploration & Par distance croissante & En profondeur d'abord \\
Sortie principale & Distances $v.d$ depuis $s$ & Horodatages $v.d$, $v.f$ \\
Complexité & $O(V+E)$ & $\Theta(V+E)$ \\
Atteint tout le graphe & Non (depuis $s$ seulement) & Oui (si on boucle sur tous les sommets) \\
Applications & Plus court chemin (non pondéré) & Tri topologique, composantes connexes \\
\bottomrule
\end{tabular}
\end{center}
