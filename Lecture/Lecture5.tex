\subsection{Matrix Multiplication}

\textbf{Problem:} given two $n \times n$ matrices $A$ and $B$, compute
$C = A \cdot B$ where $c_{ij} = \sum_{k=1}^n a_{ik} b_{kj}$.

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcll}
\toprule
\textbf{Algorithm} & \textbf{Recurrence} & \textbf{Complexity} & \textbf{Remark} \\
\midrule
Naive (3 nested loops)
  & --- & $\Theta(n^3)$ & Reference \\
Simple D\&C (8 subproblems)
  & $T(n) = 8T(n/2) + \Theta(n^2)$
  & $\Theta(n^3)$ & No improvement \\
\textbf{Strassen}
  & $T(n) = 7T(n/2) + \Theta(n^2)$
  & $\Theta(n^{\log_2 7}) \approx \Theta(n^{2.81})$ & Better! \\
\bottomrule
\end{tabular}
\end{center}

Strassen's idea: use clever algebraic combinations to perform only
\textbf{7 multiplications} of submatrices instead of 8, at the cost of a few
additional (less expensive) additions.

\begin{tcolorbox}[colback=lightblue, colframe=keyblue, fonttitle=\bfseries,
                  title={Applying the Master Theorem to Strassen}]
$a=7$, $b=2$, $f(n)=\Theta(n^2)$, $n^{\log_2 7} \approx n^{2.81}$.\\
Since $f(n) = O(n^{\log_2 7 - \varepsilon})$ (case 1):
$T(n) = \Theta(n^{\log_2 7}) \approx \Theta(n^{2.81})$.
\end{tcolorbox}

\subsubsection{Binary Heap Structure}

A \textbf{max-heap} is a \emph{nearly complete} binary tree stored in an array.

For a node at index $i$:
\[
  \textsc{Parent}(i) = \lfloor i/2 \rfloor, \quad
  \textsc{Left}(i) = 2i, \quad
  \textsc{Right}(i) = 2i+1.
\]

\textbf{Max-heap property:} $A[\textsc{Parent}(i)] \ge A[i]$ for all $i \ne$ root.
The maximum is always at the root $A[1]$.

% ─────────────────────────────────────────────────────────────────────────────
\subsubsection{Key Operations}

\paragraph{\textsc{Max-Heapify}$(A, i)$.}
Restores the max-heap property for the subtree rooted at $i$, assuming that
the left and right subtrees are already max-heaps. The element "drifts" downward.
Complexity: $O(\log n)$ (heap height).

\paragraph{\textsc{Build-Max-Heap}$(A)$.}
Builds a max-heap from an arbitrary array by calling
\textsc{Max-Heapify} on all non-leaf nodes (from $\lfloor n/2 \rfloor$ down to 1).

\begin{algorithm}[H]
\caption{Build-Max-Heap($A$)}
\begin{algorithmic}[1]
\State $A.heap\_size \leftarrow A.length$
\For{$i \leftarrow \lfloor A.length/2 \rfloor$ \textbf{downto} $1$}
    \State \Call{Max-Heapify}{$A, i$}
\EndFor
\end{algorithmic}
\end{algorithm}

\begin{tcolorbox}[colback=lightyellow, colframe=orange!70!black, fonttitle=\bfseries,
                  title={Complexity of Build-Max-Heap}]
Although we call a $O(\log n)$ function $O(n)$ times, a careful analysis
(the majority of nodes have a low height) yields:
\[ \boxed{T_{\text{build}} = \Theta(n)} \]
\end{tcolorbox}
