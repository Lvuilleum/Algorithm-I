\section*{Plus Longue Sous-Séquence Commune (LCS)}
\addcontentsline{toc}{subsection}{Plus Longue Sous-Séquence Commune (LCS)}
% ══════════════════════════════════════════════════════════════════════════════

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Définition et motivation}

\begin{definition}{Plus Longue Sous-Séquence Commune}{lcs}
\textbf{Entrée :} deux séquences $X = \langle x_1, \ldots, x_m \rangle$ et
$Y = \langle y_1, \ldots, y_n \rangle$.

\textbf{Sortie :} une sous-séquence commune aux deux dont la longueur est maximale.

\textbf{Rappel :} une sous-séquence n'a pas besoin d'être consécutive, mais doit
respecter l'ordre des éléments.
\end{definition}

\textbf{Exemple :} sur les mots \texttt{heroically} et \texttt{scholarly}, la LCS
est \texttt{holly} (longueur~5) : les lettres apparaissent dans le bon ordre dans
chacun des deux mots, sans forcément être adjacentes.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Pourquoi les approches naïves échouent}

\textbf{Force brute :} pour chaque sous-séquence de $X$, vérifier si elle est
une sous-séquence de $Y$.

\begin{complexity}{Force brute pour LCS}{lcsbf}
\begin{itemize}[noitemsep]
  \item $X$ possède $2^m$ sous-séquences.
  \item Chaque vérification prend $\Theta(n)$ (parcourir $Y$ lettre par lettre).
  \item \textbf{Complexité totale : $\Theta(n \cdot 2^m)$ — exponentielle, inutilisable.}
\end{itemize}
\end{complexity}

Il n'existe pas non plus d'algorithme glouton naturel pour ce problème :
la lettre localement la meilleure ne mène pas nécessairement à la solution globale
optimale. La programmation dynamique est le bon outil.

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Intuition de l'approche DP}

On parcourt les deux séquences depuis la \emph{droite} (la fin) vers la gauche,
étape par étape.

À chaque position $(i, j)$ (on regarde $x_i$ et $y_j$) :
\begin{itemize}
  \item \textbf{Si $x_i = y_j$} : cette lettre fait partie de la LCS ; on l'inclut
        et on résout le sous-problème sur les préfixes $X_{i-1}$ et $Y_{j-1}$.
  \item \textbf{Si $x_i \ne y_j$} : la LCS optimale s'obtient en avançant d'un pas
        à gauche dans \emph{l'une} des deux séquences, et on prend le meilleur des
        deux résultats.
\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Sous-structure optimale}

Notons $X_i = \langle x_1, x_2, \ldots, x_i \rangle$ et
$Y_j = \langle y_1, y_2, \ldots, y_j \rangle$ les préfixes de longueur $i$ et $j$.

\begin{theorem}{Sous-structure optimale pour LCS}{lcsopt}
Soit $Z = \langle z_1, z_2, \ldots, z_k \rangle$ une LCS de $X_i$ et $Y_j$.
\begin{enumerate}
  \item \textbf{Si $x_i = y_j$} : alors $z_k = x_i = y_j$ et $Z_{k-1}$ est une
        LCS de $X_{i-1}$ et $Y_{j-1}$.
  \item \textbf{Si $x_i \ne y_j$ et $z_k \ne x_i$} : alors $Z$ est une LCS de
        $X_{i-1}$ et $Y_j$.
  \item \textbf{Si $x_i \ne y_j$ et $z_k \ne y_j$} : alors $Z$ est une LCS de
        $X_i$ et $Y_{j-1}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Cas 1 ($x_i = y_j$).}
Si $z_k \ne x_i = y_j$, alors $Z' = \langle z_1, \ldots, z_k, x_i \rangle$ est une
sous-séquence commune de $X_i$ et $Y_j$ de longueur $k+1$, ce qui contredit
l'optimalité de $Z$. Donc $z_k = x_i = y_j$.

Supposons maintenant que $Z_{k-1}$ n'est pas une LCS de $X_{i-1}$ et $Y_{j-1}$ :
il existerait une sous-séquence commune $W$ de $X_{i-1}$ et $Y_{j-1}$ de
longueur $\ge k$. Alors $\langle W, z_k \rangle$ serait une sous-séquence commune
de $X_i$ et $Y_j$ de longueur $\ge k+1$, contradiction.

\textbf{Cas 2 ($x_i \ne y_j$, $z_k \ne x_i$).}
$Z$ est déjà une sous-séquence commune de $X_{i-1}$ et $Y_j$. Si elle n'était pas
optimale, il existerait $W$ de longueur $> k$ sous-séquence de $X_{i-1}$ et $Y_j$,
qui serait aussi une sous-séquence de $X_i$ et $Y_j$ (car $W$ ne finit pas par $x_i$),
contradisant l'optimalité de $Z$. Le cas~3 est symétrique.
\end{proof}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Formulation récursive}

On déduit du théorème précédent que la longueur de la LCS de $X_i$ et $Y_j$ vaut :

\[
  c[i, j] = \text{longueur de la LCS de } X_i \text{ et } Y_j
\]

avec la récurrence :

\[
  c[i, j] =
  \begin{cases}
    0 & \text{si } i = 0 \text{ ou } j = 0, \\
    c[i-1, j-1] + 1 & \text{si } i, j > 0 \text{ et } x_i = y_j, \\
    \max\bigl(c[i-1, j],\; c[i, j-1]\bigr) & \text{si } i, j > 0 \text{ et } x_i \ne y_j.
  \end{cases}
\]

On veut calculer $c[m, n]$.

\begin{remark}{Sous-problèmes répétés}{lcsov}
Une récursion naïve résoudrait les mêmes sous-problèmes un nombre exponentiel de fois.
L'approche bottom-up remplit la table $c$ une seule fois, en traitant les
sous-problèmes par taille croissante (lignes $i$ de $0$ à $m$, colonnes $j$ de $0$ à $n$).
\end{remark}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Algorithme bottom-up et exemple complet}

Pour $X = \langle B,A,B,D,B,A \rangle$ et $Y = \langle D,A,C,B,C,B,A \rangle$
($m = 6$, $n = 7$), voici la table $c[i,j]$ complète :

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{c|c|ccccccc}
& & \multicolumn{7}{c}{$j$} \\
$i$ & & \textbf{0} & \textbf{1 (D)} & \textbf{2 (A)} & \textbf{3 (C)} &
         \textbf{4 (B)} & \textbf{5 (C)} & \textbf{6 (B)} & \textbf{7 (A)} \\
\hline
\textbf{0}      & & 0 & 0 & 0 & 0 & 0 & 0 & 0 & 0 \\
\textbf{1 (B)}  & & 0 & 0 & 0 & 0 & 1 & 1 & 1 & 1 \\
\textbf{2 (A)}  & & 0 & 0 & 1 & 1 & 1 & 1 & 1 & 2 \\
\textbf{3 (B)}  & & 0 & 0 & 1 & 1 & 2 & 2 & 2 & 2 \\
\textbf{4 (D)}  & & 0 & 1 & 1 & 1 & 2 & 2 & 2 & 2 \\
\textbf{5 (B)}  & & 0 & 1 & 1 & 1 & 2 & 2 & 3 & 3 \\
\textbf{6 (A)}  & & 0 & 1 & 2 & 2 & 2 & 2 & 3 & \cellcolor{blue!20}\textbf{4} \\
\end{tabular}
\end{center}

La LCS a longueur \textbf{4} (valeur $c[6,7]$). En remontant les décisions, on
obtient la LCS \textbf{ABBA}.

\begin{remark}{Lecture de la table}{lcstable}
Pour remplir $c[i,j]$ : si $x_i = y_j$, venir de la diagonale ($c[i-1,j-1]+1$).
Sinon, prendre le maximum de la cellule du dessus ($c[i-1,j]$) et de la cellule
de gauche ($c[i,j-1]$).
\end{remark}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Pseudocode}

\begin{algorithm}[H]
\caption{LCS-Length($X$, $Y$)}
\begin{algorithmic}[1]
\State $m \leftarrow X.\mathit{length}$;\; $n \leftarrow Y.\mathit{length}$
\State Créer les tables $c[0\ldots m,\; 0\ldots n]$ et $b[1\ldots m,\; 1\ldots n]$
\For{$i = 0$ \textbf{to} $m$}
    \State $c[i, 0] \leftarrow 0$
\EndFor
\For{$j = 0$ \textbf{to} $n$}
    \State $c[0, j] \leftarrow 0$
\EndFor
\For{$i = 1$ \textbf{to} $m$}
    \For{$j = 1$ \textbf{to} $n$}
        \If{$x_i = y_j$}
            \State $c[i, j] \leftarrow c[i-1, j-1] + 1$
            \State $b[i, j] \leftarrow \text{``}\nwarrow\text{''}$   \AlgComment{diagonale : lettre en commun}
        \ElsIf{$c[i-1, j] \ge c[i, j-1]$}
            \State $c[i, j] \leftarrow c[i-1, j]$
            \State $b[i, j] \leftarrow \text{``}\uparrow\text{''}$   \AlgComment{haut : avancer dans $X$}
        \Else
            \State $c[i, j] \leftarrow c[i, j-1]$
            \State $b[i, j] \leftarrow \text{``}\leftarrow\text{''}$ \AlgComment{gauche : avancer dans $Y$}
        \EndIf
    \EndFor
\EndFor
\State \Return $c$ et $b$
\end{algorithmic}
\end{algorithm}

\begin{complexity}{LCS-Length}{lcscplx}
\begin{itemize}[noitemsep]
  \item La table $c$ contient $(m+1)(n+1)$ entrées, chacune calculée en $\Theta(1)$.
  \item \textbf{Complexité en temps : $\Theta(m \cdot n)$.}
  \item \textbf{Complexité en espace : $\Theta(m \cdot n)$.}
\end{itemize}
\end{complexity}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Reconstruction de la LCS}

La table $b$ mémorise les choix optimaux : on la remonte récursivement depuis
$b[m, n]$ vers le coin supérieur gauche.

\begin{algorithm}[H]
\caption{Print-LCS($b$, $X$, $i$, $j$)}
\begin{algorithmic}[1]
\If{$i = 0$ \textbf{or} $j = 0$}
    \State \Return
\EndIf
\If{$b[i, j] = \text{``}\nwarrow\text{''}$}
    \State \Call{Print-LCS}{$b$, $X$, $i-1$, $j-1$}
    \State \textbf{afficher} $x_i$         \AlgComment{lettre commune : on l'inclut}
\ElsIf{$b[i, j] = \text{``}\uparrow\text{''}$}
    \State \Call{Print-LCS}{$b$, $X$, $i-1$, $j$}
\Else
    \State \Call{Print-LCS}{$b$, $X$, $i$, $j-1$}
\EndIf
\end{algorithmic}
\end{algorithm}

\begin{complexity}{Print-LCS}{printlcscplx}
\begin{itemize}[noitemsep]
  \item Chaque appel récursif diminue $i + j$ d'au moins 1.
  \item La récurrence satisfait $T(n) \le T(n-1) + \Theta(1)$, d'où $T(n) = O(n)$.
  \item \textbf{Complexité totale : $\Theta(|X| + |Y|)$}
        (la borne inférieure vient de $T(n) \ge T(n-2) + \Theta(1)$).
\end{itemize}
\end{complexity}

\subsection{Résumé : LCS}

\begin{tcolorbox}[colback=lightblue, colframe=keyblue, fonttitle=\bfseries,
                  title={Récapitulatif DP – Plus Longue Sous-Séquence Commune}]
\begin{itemize}[noitemsep]
  \item \textbf{Choix} : si $x_i = y_j$, inclure cette lettre ; sinon avancer dans
        $X$ ou dans $Y$ (prendre le meilleur des deux).
  \item \textbf{Sous-structure optimale} : 3 cas selon $x_i = y_j$ ou non
        (théorème ci-dessus).
  \item \textbf{Récurrence} :
    \[
      c[i,j] =
      \begin{cases}
        0 & i = 0 \text{ ou } j = 0,\\
        c[i-1,j-1]+1 & x_i = y_j,\\
        \max(c[i-1,j],\,c[i,j-1]) & x_i \ne y_j.
      \end{cases}
    \]
  \item \textbf{Complexité} : $\Theta(mn)$ en temps et en espace.
  \item \textbf{Reconstruction} : table $b$ + \textsc{Print-LCS} en $\Theta(|X|+|Y|)$.
\end{itemize}
\end{tcolorbox}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Bilan général de la programmation dynamique}

\begin{tcolorbox}[colback=lightyellow, colframe=orange!70!black, fonttitle=\bfseries,
                  title={Recette DP en trois étapes}]
\begin{enumerate}
  \item \textbf{Identifier les choix et la sous-structure optimale.}
  \item \textbf{Écrire la valeur optimale comme récurrence sur des sous-problèmes.}
  \item \textbf{Résoudre efficacement} : top-down avec mémoïsation ou bottom-up
        (sans recalculer les mêmes sous-problèmes).
\end{enumerate}
\end{tcolorbox}
