\subsection{Reminder: Key Elements of a DP Algorithm}

Dynamic programming relies on two fundamental properties:

\begin{enumerate}
  \item \textbf{Optimal substructure}: an optimal solution is obtained by making
        a \emph{choice}, which leaves one or more subproblems to be solved,
        and the global optimal solution solves these subproblems optimally.
  \item \textbf{Overlapping subproblems}: a naive recursive algorithm recalculates the same
        subproblems many times.
        \begin{itemize}
          \item \textbf{Top-down with memoization}: solve recursively by
                storing each result in a table.
          \item \textbf{Bottom-up}: sort the subproblems and solve the smallest
                ones first; thus, when a subproblem is solved, the
                subproblems it depends on are already computed.
        \end{itemize}
\end{enumerate}

\subsection{Rod Cutting – Reconstruction of the Optimal Solution}

In Lecture~8, the DP algorithm returned only the optimal revenue $r(n)$.
It is often necessary to retrieve \emph{how} to obtain this revenue, i.e.,
the lengths of the pieces to be cut.

\subsubsection{Extended Algorithm with Decision Table}

The idea is to maintain an auxiliary table $s[0\ldots n]$: $s[j]$ stores the length
of the first piece cut (the \emph{decision}) corresponding to the optimal revenue $r[j]$.

\begin{algorithm}[H]
\caption{Extended-Bottom-Up-Cut-Rod($p$, $n$)}
\begin{algorithmic}[1]
\State Create arrays $r[0\ldots n]$ and $s[0\ldots n]$
\State $r[0] \leftarrow 0$
\For{$j = 1$ \textbf{to} $n$}
    \State $q \leftarrow -\infty$
    \For{$i = 1$ \textbf{to} $j$}
        \If{$q < p[i] + r[j - i]$}
            \State $q \leftarrow p[i] + r[j - i]$  \AlgComment{best revenue}
            \State $s[j] \leftarrow i$              \AlgComment{store the optimal choice}
        \EndIf
    \EndFor
    \State $r[j] \leftarrow q$
\EndFor
\State \Return $r$ and $s$
\end{algorithmic}
\end{algorithm}

The complexity remains $\Theta(n^2)$. The only difference from the basic algorithm
is the maintenance of $s[j]$ which records the index $i$ achieving the maximum.

\subsubsection{Example: Tables $r$ and $s$ for $n = 8$}

Using the price table $p_i$: 1, 5, 8, 9, 10, 17, 17, 20, 24, 30:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{ccccccccccc}
\toprule
$i$    & 0 & 1 & 2 & 3 & 4  & 5  & 6  & 7  & 8  \\
\midrule
$r[i]$ & 0 & 1 & 5 & 8 & 10 & 13 & 17 & 18 & 22 \\
$s[i]$ & 0 & 1 & 2 & 3 & 2  & 2  & 6  & 1  & 2  \\
\bottomrule
\end{tabular}
\end{center}

\subsubsection{Extraction of the Solution}

Table $s$ stores the choices leading to an optimal solution.
We reconstruct the solution by tracing back from $n$:

\begin{algorithm}[H]
\caption{Print-Cut-Rod-Solution($p$, $n$)}
\begin{algorithmic}[1]
\State $(r, s) \leftarrow \textsc{Extended-Bottom-Up-Cut-Rod}(p, n)$
\While{$n > 0$}
    \State \textbf{print} $s[n]$   \AlgComment{size of the next piece}
    \State $n \leftarrow n - s[n]$
\EndWhile
\end{algorithmic}
\end{algorithm}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Summary: Rod Cutting}

\begin{tcolorbox}[colback=lightblue, colframe=keyblue, fonttitle=\bfseries,
                  title={Summary DP – Rod Cutting}]
\begin{itemize}[noitemsep]
  \item \textbf{Choice}: where to make the leftmost cut.
  \item \textbf{Optimal substructure}: the remainder must be cut optimally.
  \item \textbf{Recurrence}:
    \[
      r(n) =
      \begin{cases}
        0 & \text{si } n = 0, \\
        \displaystyle\max_{1 \le i \le n} \bigl\{p_i + r(n-i)\bigr\} & \text{si } n \ge 1.
      \end{cases}
    \]
  \item \textbf{Complexity}: $\Theta(n^2)$ (top-down or bottom-up).
  \item \textbf{Reconstruction}: auxiliary table $s$ + trace-back in $O(n)$.
\end{itemize}
\end{tcolorbox}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{When Can Dynamic Programming Be Used?}

\begin{enumerate}
  \item \textbf{Optimal substructure.}
    \begin{itemize}
      \item An optimal solution can be constructed by combining optimal solutions
            of subproblems.
      \item This implies that the optimal value can be expressed by a
            \emph{recursive formula}.
    \end{itemize}
  \item \textbf{Overlapping subproblems.}
    \begin{itemize}
      \item A naive recursive algorithm may revisit the same subproblem
            many times.
    \end{itemize}
\end{enumerate}

\begin{remark}{Coin Change Problem}{change9}
DP applies directly to this classic problem. With $n$ denominations
$0 < w_1 < \cdots < w_n$ and a target amount $W$, we minimize the number of coins:
\[
  \min\left\{\sum_{j=1}^{n} x_j \;:\; \sum_{j=1}^{n} w_j x_j = W,\; x_j \in \mathbb{Z}_{\ge 0}\right\}.
\]
\textbf{Example:} $w_1=1$, $w_2=2$, $w_3=5$, $W=8$.
Optimal: $x_1 = x_2 = x_3 = 1 \Rightarrow$ 3~coins.
\end{remark}

\end{tikzpicture}

Each entry of $C$ requires $q$ scalar multiplications.
The matrix $C$ contains $p \cdot r$ entries, so the total cost is
$\mathbf{p \cdot q \cdot r}$ \textbf{scalar multiplications}.

\begin{tcolorbox}[colback=lightyellow, colframe=orange!70!black, fonttitle=\bfseries,
                  title={Why does the association order matter?}]
For $A$ ($20 \times 1$), $B$ ($1 \times 10$), $C$ ($10 \times 100$):
\begin{itemize}[noitemsep]
  \item $(A \times (B \times C))$ : $1\cdot10\cdot100 + 20\cdot1\cdot100 = 3,000$
  \item $((A \times B) \times C)$ : $20\cdot1\cdot10 + 20\cdot10\cdot100 = 20,200$
\end{itemize}
Parenthesization can \textbf{radically modify} the number of operations.
\end{tcolorbox}

\subsection{Problem Definition}

\begin{definition}{Matrix Chain Multiplication}{mcm}
\textbf{Input:} a sequence $\langle A_1, A_2, \ldots, A_n \rangle$ of $n$ matrices,
where matrix $A_i$ has dimension $p_{i-1} \times p_i$ for $i = 1, \ldots, n$.

\textbf{Output:} a complete parenthesization of the product $A_1 A_2 \cdots A_n$
minimizing the number of scalar multiplications.
\end{definition}

\textbf{Notes:}
\begin{itemize}
  \item We do \emph{not} ask to compute the product, only to find the
        best parenthesization.
  \item Parenthesization can considerably affect the number of
        multiplications.
\end{itemize}

\textbf{Example:} sequence $A_1 A_2 A_3$ with dimensions $50\times5$, $5\times100$,
$100\times10$:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lrl}
\toprule
\textbf{Parenthesization} & \textbf{Cost} & \textbf{Calculation} \\
\midrule
$(A_1 A_2) A_3$ & $75,000$ & $50\cdot5\cdot100 + 50\cdot100\cdot10$ \\
$A_1 (A_2 A_3)$ & $7,500$  & $5\cdot100\cdot10 + 50\cdot5\cdot10$ \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Optimal Substructure}

\begin{theorem}{Optimal Substructure – Matrix Multiplication}{mcmopt}
Suppose the optimal parenthesization of $A_1 A_2 \cdots A_n$ performs the
last multiplication as $(A_1 \cdots A_k)(A_{k+1} \cdots A_n)$. Then the
parenthesizations of $A_1 \cdots A_k$ and $A_{k+1} \cdots A_n$ in this
optimal solution are themselves optimal.
\end{theorem}

\begin{proof}
Let $M(P)$ be the number of scalar multiplications of a parenthesization $P$.
Let $((O_L) \cdot (O_R))$ be the optimal solution where $O_L$ and $O_R$ parenthesize
respectively $A_1\cdots A_k$ and $A_{k+1}\cdots A_n$. Let $P_L$ and $P_R$ be
optimal parenthesizations of these two subchains. We have:
\begin{align*}
  M((O_L) \cdot (O_R))
    &= p_0 \cdot p_k \cdot p_n + M(O_L) + M(O_R) \\
    &\ge p_0 \cdot p_k \cdot p_n + M(P_L) + M(P_R)
     = M((P_L) \cdot (P_R)).
\end{align*}
Since $((O_L)\cdot(O_R))$ is optimal, we have equality, thus
$M(P_L) = M(O_L)$ and $M(P_R) = M(O_R)$: $P_L$ and $P_R$ are indeed optimal.
\end{proof}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Recursive Formula}

Let $m[i, j]$ be the minimum number of scalar multiplications to compute
$A_i A_{i+1} \cdots A_j$.

\[
  m[i, j] =
  \begin{cases}
    0 & \text{if } i = j, \\
    \displaystyle\min_{i \le k < j}
      \bigl\{ m[i,k] + m[k+1,j] + p_{i-1}\, p_k\, p_j \bigr\}
    & \text{if } i < j.
  \end{cases}
\]

\begin{remark}{Filling Order}{ordrem}
$m[i, j]$ depends only on subproblems with a \emph{smaller} interval $j - i$.
A bottom-up algorithm therefore solves subproblems by increasing order of $j - i$
(chain length $\ell = j - i + 1$).
\end{remark}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Bottom-up Algorithm}

\begin{algorithm}[H]
\caption{Matrix-Chain-Order($p$)}
\begin{algorithmic}[1]
\State $n \leftarrow p.\mathit{length} - 1$
\State Create tables $m[1\ldots n,\, 1\ldots n]$ and $s[1\ldots n,\, 1\ldots n]$
\For{$i = 1$ \textbf{to} $n$}
    \State $m[i, i] \leftarrow 0$
\EndFor
\For{$\ell = 2$ \textbf{to} $n$}                  \AlgComment{$\ell$ = chain length}
    \For{$i = 1$ \textbf{to} $n - \ell + 1$}
        \State $j \leftarrow i + \ell - 1$
        \State $m[i, j] \leftarrow \infty$
        \For{$k = i$ \textbf{to} $j - 1$}
            \State $q \leftarrow m[i,k] + m[k+1,j] + p_{i-1}\, p_k\, p_j$
            \If{$q < m[i,j]$}
                \State $m[i,j] \leftarrow q$
                \State $s[i,j] \leftarrow k$    \AlgComment{$s$ stores the optimal choice}
            \EndIf
        \EndFor
    \EndFor
\EndFor
\State \Return $m$ and $s$
\end{algorithmic}
\end{algorithm}

\begin{complexity}{Matrix-Chain-Order}{mcmcplx}
\begin{itemize}[noitemsep]
  \item Table $m$ contains $O(n^2)$ entries.
  \item Each entry $m[i,j]$ requires $O(n)$ to find the best $k$.
  \item \textbf{Total complexity: $\Theta(n^3)$ in time, $\Theta(n^2)$ in space.}
\end{itemize}
\end{complexity}

\subsection{Reconstructing the Optimal Parenthesization}

Table $s$ allows for recursively reconstructing the parenthesization:

\begin{algorithm}[H]
\caption{Print-Optimal-Parens($s$, $i$, $j$)}
\begin{algorithmic}[1]
\If{$i = j$}
    \State \textbf{print} ``$A_i$''
\Else
    \State \textbf{print} ``(''
    \State \Call{Print-Optimal-Parens}{$s$, $i$, $s[i,j]$}
    \State \Call{Print-Optimal-Parens}{$s$, $s[i,j]+1$, $j$}
    \State \textbf{print} ``)''
\EndIf
\end{algorithmic}
\end{algorithm}

\subsection{Summary: Matrix Chain Multiplication}

\begin{tcolorbox}[colback=lightblue, colframe=keyblue, fonttitle=\bfseries,
                  title={Summary DP – Matrix Chain Multiplication}]
\begin{itemize}[noitemsep]
  \item \textbf{Choice}: where to place the outermost parenthesis
        $(A_1 \cdots A_k)(A_{k+1} \cdots A_n)$.
  \item \textbf{Optimal substructure}: both subchains must be
        parenthesized optimally (proven by the cut-and-paste argument).
  \item \textbf{Recurrence}:
    \[
      m[i,j] =
      \begin{cases}
        0 & \text{si } i = j, \\
        \displaystyle\min_{i \le k < j}
          \bigl\{m[i,k] + m[k+1,j] + p_{i-1}\,p_k\,p_j\bigr\}
        & \text{si } i < j.
      \end{cases}
    \]
  \item \textbf{Complexity}: $\Theta(n^3)$ in time, $\Theta(n^2)$ in space.
  \item \textbf{Reconstruction}: auxiliary table $s$ + \textsc{Print-Optimal-Parens}.
\end{itemize}
\end{tcolorbox}

The three systematic steps remain:
\begin{enumerate}
  \item Identify choices and optimal substructure.
  \item Write the recurrence for the optimal value.
  \item Solve efficiently (top-down with memoization or bottom-up).
\end{enumerate}
